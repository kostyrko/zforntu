ng-bootstrap

Instalacja

ng add @ng-bootstrap/ng-bootstrap

Setup

AppComponent import {NgbConfig} from '@ng-bootstrap/ng-bootstrap';

export class AppComponent {
  constructor(ngbConfig: NgbConfig) {
    ngbConfig.animation = false;
  }
}
